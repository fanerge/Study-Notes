function Foo( name ){
	//私有成员
	let name = 'fan';
	let getName = function(){
		console.log( name );
	};
	
	//特权成员(公有成员)
	this.name = name;
	this.getName = function(){   //可以访问私有成员
		console.log( name );  //fan
	};
}

//公有成员
Foo.prototype.name = 'yuzhenfan';
Foo.prototype.getName = function(){  //可以访问公有成员
	return this.name;
};

//静态成员
Foo.name = 'cube';
Foo.getName = function(){
	console.log();
};

总结：
1.私有成员只能在构造函数内部访问
2公有成员必须实例化后方可使用，且特权方法时可以访问私有变量，而外部特权方法时不可以访问的
3.静态成员是可以直接访问(不需要实例化)
4.this ---- 所在函数的调用方式(环境)所决定的。
补充：
1.函数声明 与 函数表达式 区别
	//函数声明
	function getName( name ){
		console.log(name);
	}
	//函数表达式
	let getName = function( name ){
		console.log(name);
	}
函数声明在JS解析时进行函数提升，因此在同一个作用域内，不管函数声明在哪里定义，该函数都可以进行调用。	
函数表达式的值是在JS运行时确定，并且在表达式赋值完成后，该函数才能调用。
